    <h2>Anh có rảnh vào chủ nhật này không? ❤️ 🥰</h2>

    <button id="agree" class="btn" onclick="acceptLove()">Có chứ</button>

    <button id="disagree" class="btn" onclick="refuseLove()">Không rảnh</button>

    <!-- Popup -->

    <div id="popup" class="popup">

        <p>Tuyệt vời! Mình cùng đi chơi nhé ❤️</p>

        <button onclick="closePopup()">OK nhaa 😘</button>

    </div>

    <script>

        let agreeBtn = document.getElementById("agree");

        let disagreeBtn = document.getElementById("disagree");

        let popup = document.getElementById("popup");

        let disagreeCount = 0;

        function acceptLove() {

            popup.style.display = "block"; // Hiện popup

        }

        function closePopup() {

            popup.style.display = "none"; // Ẩn popup

        }

        function refuseLove() {

            let messages = [

                "Em sẽ buồn đấy 😢",

                "Em buồn 😭",

                "Em buồn lắm 😔",

                "Em buồn mà hụ hụ🥺"

            ];

            // Cập nhật text theo thứ tự rồi lặp lại

            disagreeBtn.innerText = messages[disagreeCount % messages.length];

            // Tăng kích thước nút "Có chứ"

            agreeBtn.style.fontSize = (20 + disagreeCount * 5) + "px";

            // Tăng biến đếm

            disagreeCount++;

        }

    </script>
